

		import java.io.*;
		import java.util.*;
		class T5{

public static void main(String [] args){

	
		for(int hang=1;hang<=6;hang++){

			
			//打印空格	6-x个空格  
			for(int i=1;i<=(6-hang);i++){
				System.out.print(" ");

			}




				//打印星  2x-1  *
			for(int j=1;j<=(2*hang-1);j++){

				System.out.print("*");

			}

			System.out.println();


			


		}






}


}
